CSV-driven alignment generator

Reads a CSV of PIs (Point of Intersection, radius, spiral-in, spiral-out), validates against a design-criteria set, and creates the alignment programmatically. Replaces the manual PI-by-PI fixed-curve workflow.

m1_alignment_from_csv.py 
# ============================================================
# M1 — Build Civil 3D alignment from a PI CSV
# Inputs (Dynamo):
#   IN[0] : csv_path (string)
#   IN[1] : alignment_name (string)
#   IN[2] : criteria_set_name (string)  e.g. "AASHTO-Metric-80"
#   IN[3] : start_station (float)        e.g. 0.0
# ============================================================

import clr, csv
clr.AddReference('AcMgd')
clr.AddReference('AcDbMgd')
clr.AddReference('AeccDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode, Transaction
from Autodesk.AutoCAD.Geometry         import Point2d, Point3d
from Autodesk.Civil.DatabaseServices  import (
    Alignment, AlignmentType, AlignmentSubEntityType
)

csv_path           = IN[0]
alignment_name     = IN[1]
criteria_set_name  = IN[2]
start_station      = float(IN[3] if IN[3] else 0.0)

def read_pis(path):
    pis = []
    with open(path, 'rb') as f:
        reader = csv.DictReader(f)
        for row in reader:
            pis.append({
                'id'     : row['pi_id'],
                'pt'     : Point2d(float(row['easting']), float(row['northing'])),
                'radius' : float(row['radius_m']),
                'spi'    : float(row['spiral_in_m']),
                'spo'    : float(row['spiral_out_m']),
                'speed'  : float(row['design_speed_kph']),
            })
    return pis


def validate(pis):
    # Simple QA: min radius vs design speed (placeholder rule)
    issues = []
    for p in pis:
        if p['radius'] > 0 and p['radius'] < p['speed'] * 2.5:
            issues.append("PI %s radius %.1f below 2.5*V threshold" % (p['id'], p['radius']))
    return issues


def build_alignment():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = Autodesk.Civil.ApplicationServices.CivilApplication.ActiveDocument

    pis = read_pis(csv_path)
    issues = validate(pis)

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            # Resolve layer + styles (already configured in template DWG)
            site_id  = civdoc.GetSiteIds()[0] if civdoc.GetSiteIds().Count > 0 else Autodesk.AutoCAD.DatabaseServices.ObjectId.Null
            style_id = civdoc.Styles.AlignmentStyles["Standard"]
            label_id = civdoc.Styles.LabelSetStyles.AlignmentLabelSetStyles["Standard"]

            align_id = Alignment.Create(
                civdoc, alignment_name, site_id,
                "0", style_id, label_id,
                AlignmentType.Centerline
            )
            align = tr.GetObject(align_id, OpenMode.ForWrite)
            align.StationIndexingIncrement = 100.0
            align.ReferenceStation         = start_station
            align.DesignSpeedSummary.Clear()

            entities = align.Entities
            prev = None
            for i, pi in enumerate(pis):
                if i == 0:
                    prev = pi['pt']
                    continue
                if pi['radius'] == 0 or i == len(pis)-1:
                    entities.AddFixedLine(prev, pi['pt'])
                else:
                    nxt = pis[i+1]['pt'] if i+1 < len(pis) else pi['pt']
                    if pi['spi'] > 0 and pi['spo'] > 0:
                        entities.AddFreeSpiralCurveSpiral(
                            prev, pi['pt'], nxt,
                            pi['spi'], pi['radius'], pi['spo']
                        )
                    else:
                        entities.AddFreeCurve(prev, pi['pt'], nxt, pi['radius'])
                prev = pi['pt']

            # Stamp design speed across whole alignment
            align.DesignSpeedSummary.AddDesignSpeed(start_station, pis[0]['speed'])

            tr.Commit()

    msg = "OK: alignment '%s' created with %d PIs" % (alignment_name, len(pis))
    if issues:
        msg += " | WARNINGS: " + "; ".join(issues)
    return msg

OUT = build_alignment()
Why this lifts automation: the manual PI-by-PI workflow with curve resolution typically takes 15–30 min per alignment. This script reduces it to seconds, and QA flags are produced automatically.

Python script

m2_profile_from_csv.py 
# ============================================================
# M2 — Build finished-grade profile from PVI CSV
# Inputs (Dynamo):
#   IN[0] : csv_path
#   IN[1] : alignment_name (target)
#   IN[2] : profile_name           e.g. "FG"
#   IN[3] : max_grade_pct          e.g. 6.0
#   IN[4] : min_grade_pct          e.g. 0.5
# ============================================================

import clr, csv
clr.AddReference('AcDbMgd'); clr.AddReference('AcMgd'); clr.AddReference('AeccDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode
from Autodesk.Civil.DatabaseServices  import (
    Alignment, Profile, ProfileType, VerticalCurveType
)
from Autodesk.Civil.ApplicationServices import CivilApplication

csv_path        = IN[0]
align_name      = IN[1]
profile_name    = IN[2]
max_grade_pct   = float(IN[3])
min_grade_pct   = float(IN[4])


def read_pvis(path):
    rows = []
    with open(path, 'rb') as f:
        for r in csv.DictReader(f):
            rows.append({
                'sta'  : float(r['station_m']),
                'elev' : float(r['elevation_m']),
                'kc'   : float(r['k_crest']),
                'ks'   : float(r['k_sag']),
                'len'  : float(r['curve_length_m']),
            })
    rows.sort(key=lambda x: x['sta'])
    return rows


def grade_qa(pvis):
    issues = []
    for i in range(1, len(pvis)):
        d = pvis[i]['sta'] - pvis[i-1]['sta']
        if d <= 0: continue
        g = (pvis[i]['elev'] - pvis[i-1]['elev']) / d * 100.0
        if abs(g) > max_grade_pct:
            issues.append("Grade %.2f%% between %.1f-%.1f exceeds max %.1f%%"
                          % (g, pvis[i-1]['sta'], pvis[i]['sta'], max_grade_pct))
        if 0 < abs(g) < min_grade_pct:
            issues.append("Grade %.2f%% below drainage min %.1f%%" % (g, min_grade_pct))
    return issues


def build_profile():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = CivilApplication.ActiveDocument

    pvis   = read_pvis(csv_path)
    issues = grade_qa(pvis)

    align_id = None
    for aid in civdoc.GetAlignmentIds():
        align_id = aid
        if Alignment.GetAlignmentName(aid) == align_name: break
    if align_id is None:
        return "ERROR: alignment '%s' not found" % align_name

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            align = tr.GetObject(align_id, OpenMode.ForWrite)
            style_id = civdoc.Styles.ProfileStyles["Design Profile"]
            label_id = civdoc.Styles.LabelSetStyles.ProfileLabelSetStyles["Standard"]

            prof_id = Profile.CreateByLayout(
                profile_name, align_id, "0", style_id, label_id
            )
            prof = tr.GetObject(prof_id, OpenMode.ForWrite)
            ents = prof.Entities

            # Add tangents
            for i in range(1, len(pvis)):
                a, b = pvis[i-1], pvis[i]
                ents.AddFixedTangent(a['sta'], a['elev'], b['sta'], b['elev'])

            # Add vertical curves at internal PVIs
            for i in range(1, len(pvis)-1):
                p = pvis[i]
                if p['len'] <= 0: continue
                ents.AddFreeVerticalCurveByPVILength(
                    p['sta'], p['len'], VerticalCurveType.Parabolic
                )

            tr.Commit()

    msg = "OK: profile '%s' built with %d PVIs" % (profile_name, len(pvis))
    if issues:
        msg += " | QA: " + "; ".join(issues)
    return msg

OUT = build_profile()
Engineering judgement preserved: the script doesn't choose K-values for you — it reads them from the CSV (or your firm's K-table). You still review the QA output before the next stage.

# ============================================================
# M3 — Apply width table + bind targets across corridor regions
# Inputs:
#   IN[0] : csv_path
#   IN[1] : corridor_name
#   IN[2] : assembly_name (existing assembly with named subassemblies:
#           'Lane_L', 'Lane_R', 'Shoulder_L', 'Shoulder_R')
# ============================================================

import clr, csv
clr.AddReference('AeccDbMgd'); clr.AddReference('AcMgd')
from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode
from Autodesk.Civil.DatabaseServices import (
    Corridor, Assembly, Subassembly, SubassemblyTargetType
)
from Autodesk.Civil.ApplicationServices import CivilApplication

csv_path       = IN[0]
corridor_name  = IN[1]
assembly_name  = IN[2]


def load_table(path):
    with open(path, 'rb') as f:
        return list(csv.DictReader(f))


def find_feature_line_id(civdoc, name):
    for site_id in civdoc.GetSiteIds():
        site = civdoc.GetSite(site_id)
        for fl_id in site.GetFeatureLineIds():
            if Autodesk.Civil.DatabaseServices.FeatureLine.GetFeatureLineName(fl_id) == name:
                return fl_id
    return None


def apply_widths_and_targets():
    doc    = Application.DocumentManager.MdiActiveDocument
    db     = doc.Database
    civdoc = CivilApplication.ActiveDocument

    rows = load_table(csv_path)

    # Find corridor by name
    cor_id = None
    for cid in civdoc.CorridorCollection.GetCorridorIds():
        if Corridor.GetCorridorName(cid) == corridor_name:
            cor_id = cid; break
    if cor_id is None:
        return "ERROR: corridor '%s' not found" % corridor_name

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            cor = tr.GetObject(cor_id, OpenMode.ForWrite)
            baseline = cor.Baselines[0]

            # Clear existing regions, recreate from table
            baseline.BaselineRegions.Clear()
            for r in rows:
                region = baseline.BaselineRegions.Add(
                    r['region_id'], float(r['start_sta']), float(r['end_sta'])
                )
                region.Frequency.ApplyTangentRunoff      = True
                region.Frequency.MidOrdinateDistance     = 0.05

                # Set width params on named subassemblies
                for sa_id in region.AssemblyId.Subassemblies:
                    sa = tr.GetObject(sa_id, OpenMode.ForWrite)
                    name = sa.Name
                    if name == "Lane_L" or name == "Lane_R":
                        sa.ParamsLong["Width"].Value = float(r['lane_width_m'])
                    elif name == "Shoulder_L":
                        sa.ParamsLong["Width"].Value = float(r['shoulder_l_m'])
                    elif name == "Shoulder_R":
                        sa.ParamsLong["Width"].Value = float(r['shoulder_r_m'])

                # Bind offset targets by name
                t_left  = find_feature_line_id(civdoc, r['target_l'])
                t_right = find_feature_line_id(civdoc, r['target_r'])
                for tgt in region.GetTargets():
                    if tgt.TargetType != SubassemblyTargetType.Width: continue
                    if "_L" in tgt.SubassemblyName and t_left:
                        tgt.SetTargetIds([t_left])
                    elif "_R" in tgt.SubassemblyName and t_right:
                        tgt.SetTargetIds([t_right])

            tr.Commit()

    return "OK: %d regions bound, widths set, targets resolved" % len(rows)

OUT = apply_widths_and_targets()

import clr, csv
clr.AddReference('AeccDbMgd'); clr.AddReference('AcMgd'); clr.AddReference('AcDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode
from Autodesk.Civil.DatabaseServices import (
    Corridor, TinSurface, TinVolumeSurface
)
from Autodesk.Civil.ApplicationServices import CivilApplication

corridor_name      = IN[0]
eg_surface_name    = IN[1]
fg_surface_name    = IN[2]
out_csv_path       = IN[3]
station_interval_m = float(IN[4] if IN[4] else 20.0)


def run():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = CivilApplication.ActiveDocument

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            cor = None
            for cid in civdoc.CorridorCollection.GetCorridorIds():
                if Corridor.GetCorridorName(cid) == corridor_name:
                    cor = tr.GetObject(cid, OpenMode.ForWrite); break
            if cor is None:
                return "ERROR: corridor not found"

            cor.Rebuild()

            # Ensure corridor surface exists
            if cor.CorridorSurfaces.Count == 0:
                cs = cor.CorridorSurfaces.Add(fg_surface_name)
                cs.SurfaceDefinition.AddLinks("Top")
            cor.Rebuild()

            # Get TIN ids
            eg_id = fg_id = None
            for sid in civdoc.GetSurfaceIds():
                s = tr.GetObject(sid, OpenMode.ForRead)
                if s.Name == eg_surface_name: eg_id = sid
                if s.Name == fg_surface_name: fg_id = sid

            # Create / refresh volume surface
            vol_name = fg_surface_name + "_VOL_vs_EG"
            vol_id   = TinVolumeSurface.Create(civdoc, vol_name,
                          civdoc.Styles.SurfaceStyles["Standard"],
                          civdoc.Styles.LabelSetStyles.SurfaceLabelSetStyles["Standard"],
                          eg_id, fg_id)
            vs = tr.GetObject(vol_id, OpenMode.ForRead)
            cut  = vs.GetVolumeProperties().UnadjustedCutVolume
            fill = vs.GetVolumeProperties().UnadjustedFillVolume
            net  = cut - fill

            # Write summary CSV
            with open(out_csv_path, 'wb') as f:
                w = csv.writer(f)
                w.writerow(["corridor", "cut_m3", "fill_m3", "net_m3"])
                w.writerow([corridor_name, "%.2f"%cut, "%.2f"%fill, "%.2f"%net])

            tr.Commit()

    return "OK: rebuilt + volumes  cut=%.2f fill=%.2f net=%.2f" % (cut, fill, net)

OUT = run()

import clr, csv, math
clr.AddReference('AcMgd'); clr.AddReference('AcDbMgd'); clr.AddReference('AeccDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import (
    OpenMode, BlockTable, BlockTableRecord, BlockReference
)
from Autodesk.AutoCAD.Geometry         import Point3d, Vector3d
from Autodesk.Civil.DatabaseServices  import Alignment
from Autodesk.Civil.ApplicationServices import CivilApplication

csv_path     = IN[0]
align_name   = IN[1]
layer_name   = IN[2]   # e.g. "C-SGN-FURN"


def station_to_point(align, sta, off, side):
    sign = -1.0 if side.upper() == "L" else 1.0
    pt = align.PointLocation(sta, sign * off)
    ang = align.GetTangentAngle(sta)
    return Point3d(pt.X, pt.Y, 0), ang


def place_signs():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = CivilApplication.ActiveDocument

    rows = list(csv.DictReader(open(csv_path, 'rb')))

    align_id = None
    for aid in civdoc.GetAlignmentIds():
        if Alignment.GetAlignmentName(aid) == align_name:
            align_id = aid; break
    if align_id is None: return "ERROR: alignment not found"

    placed, missing = 0, []

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            bt  = tr.GetObject(db.BlockTableId, OpenMode.ForRead)
            msp = tr.GetObject(bt[BlockTableRecord.ModelSpace], OpenMode.ForWrite)
            align = tr.GetObject(align_id, OpenMode.ForRead)

            for r in rows:
                blk_name = r['block_name']
                if not bt.Has(blk_name):
                    missing.append(blk_name); continue

                pt, ang = station_to_point(
                    align,
                    float(r['station_m']),
                    float(r['offset_m']),
                    r['side']
                )
                rot = math.radians(float(r['rotation_deg'])) if r['rotation_deg'] else ang
                br  = BlockReference(pt, bt[blk_name])
                br.Rotation = rot
                br.Layer    = layer_name
                msp.AppendEntity(br)
                tr.AddNewlyCreatedDBObject(br, True)
                placed += 1

            tr.Commit()

    msg = "OK: %d signs placed" % placed
    if missing:
        msg += " | missing blocks: " + ", ".join(sorted(set(missing)))
    return msg

OUT = place_signs()

# M6 — Generate view frames + publish PDFs
# Inputs:
#   IN[0] : alignment_name

import clr, os, datetime, yaml
clr.AddReference('AcMgd'); clr.AddReference('AcDbMgd'); clr.AddReference('AeccDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode
from Autodesk.Civil.DatabaseServices import (
    Alignment, ViewFrameGroup, ViewFrameGroupCreationData
)
from Autodesk.Civil.ApplicationServices import CivilApplication

align_name      = IN[0]
template_dwt    = IN[1]
page_setup_name = IN[2]
sheet_len       = float(IN[3])
scale           = float(IN[4])
header_yaml     = IN[5]
output_dir      = IN[6]


def run():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = CivilApplication.ActiveDocument
    header = yaml.safe_load(open(header_yaml, 'rb'))

    align_id = None
    for aid in civdoc.GetAlignmentIds():
        if Alignment.GetAlignmentName(aid) == align_name:
            align_id = aid; break
    if align_id is None: return "ERROR: alignment not found"

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            data = ViewFrameGroupCreationData(
                "VFG_" + align_name, align_id,
                template_dwt, page_setup_name
            )
            data.SheetLength = sheet_len
            data.Scale       = scale
            data.MatchLineOverlap = 5.0
            vfg_id = ViewFrameGroup.Create(civdoc, data)

            # Field substitution stub — real SSM handled via .NET
            for k, v in header.items():
                db.UpdateFieldVariable(k, str(v))

            tr.Commit()

    # Publish using AutoCAD PublishExecute — outline only
    today = datetime.date.today().isoformat()
    out   = os.path.join(output_dir, today, header.get("revision", "REV-A"))
    if not os.path.isdir(out): os.makedirs(out)

    Application.DocumentManager.MdiActiveDocument.SendStringToExecute(
        '-PUBLISH ' + out + ' ', True, False, True
    )
    return "OK: view frames built, publish queued -> " + out

OUT = run()

# ============================================================
# M7 — Roll up quantities, write Excel BOQ
# Inputs:
#   IN[0] : corridor_name
#   IN[1] : marking_layer        (e.g. "C-ROAD-MARK")
#   IN[2] : signage_layer        (e.g. "C-SGN-FURN")
#   IN[3] : payitems_csv
#   IN[4] : output_xlsx
# ============================================================

import clr, csv
clr.AddReference('AcMgd'); clr.AddReference('AcDbMgd'); clr.AddReference('AeccDbMgd')

from Autodesk.AutoCAD.ApplicationServices import Application
from Autodesk.AutoCAD.DatabaseServices import OpenMode, BlockTableRecord
from Autodesk.Civil.DatabaseServices import Corridor
from Autodesk.Civil.ApplicationServices import CivilApplication

# openpyxl is bundled with most Dynamo CPython3 setups; for IronPython
# you can write CSV here and post-process with Excel separately.
try:
    from openpyxl import Workbook
    HAVE_XLSX = True
except:
    HAVE_XLSX = False

corridor_name   = IN[0]
marking_layer   = IN[1]
signage_layer   = IN[2]
payitems_csv    = IN[3]
output_xlsx     = IN[4]


def load_paymap(path):
    m = {}
    for r in csv.DictReader(open(path, 'rb')):
        m[(r['source'], r['key'])] = r
    return m


def corridor_quantities(tr, civdoc):
    # Pull material quantities by name from corridor QTO
    out = {}
    for cid in civdoc.CorridorCollection.GetCorridorIds():
        if Corridor.GetCorridorName(cid) != corridor_name: continue
        cor = tr.GetObject(cid, OpenMode.ForRead)
        for mat in cor.GetCorridorMaterialList():
            out[mat.Name] = out.get(mat.Name, 0.0) + mat.Volume
    return out


def marking_quantities(tr, db):
    # Sum lengths by XData type stamp set in the marking module
    out = {}
    msp = tr.GetObject(
        tr.GetObject(db.BlockTableId, OpenMode.ForRead)[BlockTableRecord.ModelSpace],
        OpenMode.ForRead)
    for oid in msp:
        ent = tr.GetObject(oid, OpenMode.ForRead)
        if ent.Layer != marking_layer: continue
        xd = ent.GetXDataForApplication("ROAD_MARK_SRC")
        if xd is None: continue
        # Find mark_type tag (third TypedValue per writer in original module)
        mtype = None
        for tv in xd:
            if tv.TypeCode == 1000 and tv.Value in ("M1","M2","M3","M4","M5","M6","M7","M8","M9","M10"):
                mtype = tv.Value
        if mtype is None: continue
        # Length approximated from bounding box X-extent (or true length via solid edges)
        bb  = ent.GeometricExtents
        ln  = (bb.MaxPoint.X - bb.MinPoint.X)
        out[mtype] = out.get(mtype, 0.0) + ln
    return out


def signage_counts(tr, db):
    out = {}
    msp = tr.GetObject(
        tr.GetObject(db.BlockTableId, OpenMode.ForRead)[BlockTableRecord.ModelSpace],
        OpenMode.ForRead)
    for oid in msp:
        ent = tr.GetObject(oid, OpenMode.ForRead)
        if ent.Layer != signage_layer: continue
        if hasattr(ent, "Name"):
            # Map block name → sign code by simple prefix strip
            code = ent.Name.replace("SGN_","").replace("_","-")
            out[code] = out.get(code, 0) + 1
    return out


def run():
    doc = Application.DocumentManager.MdiActiveDocument
    db  = doc.Database
    civdoc = CivilApplication.ActiveDocument
    paymap = load_paymap(payitems_csv)
    rows = []

    with doc.LockDocument():
        with db.TransactionManager.StartTransaction() as tr:
            corr = corridor_quantities(tr, civdoc)
            marks = marking_quantities(tr, db)
            signs = signage_counts(tr, db)
            tr.Commit()

    for k, v in corr.items():
        m = paymap.get(("corridor", k));   rows.append((m, v) if m else (None, k, v))
    for k, v in marks.items():
        m = paymap.get(("marking", k));   rows.append((m, v) if m else (None, k, v))
    for k, v in signs.items():
        m = paymap.get(("signage", k));   rows.append((m, v) if m else (None, k, v))

    if HAVE_XLSX:
        wb = Workbook(); ws = wb.active; ws.title = "BOQ"
        ws.append(["Pay item","Description","Unit","Quantity"])
        for r in rows:
            if r[0] is None: continue
            m, qty = r
            ws.append([m['pay_item'], m['description'], m['unit'], round(qty,2)])
        wb.save(output_xlsx)
        return "OK: BOQ written to " + output_xlsx
    else:
        # CSV fallback for IronPython
        with open(output_xlsx.replace(".xlsx",".csv"), 'wb') as f:
            w = csv.writer(f)
            w.writerow(["Pay item","Description","Unit","Quantity"])
            for r in rows:
                if r[0] is None: continue
                m, qty = r
                w.writerow([m['pay_item'], m['description'], m['unit'], round(qty,2)])
        return "OK: BOQ CSV written (xlsx requires openpyxl)"

OUT = run()

    try:
        msg = callable_()

Operating discipline

Governance, QA & safe automation

Automation only stays useful if it stays trustworthy. These rules keep the pipeline auditable and reversible.

G1
Inputs are versioned
All CSVs and project.yaml live in git. Every drawing revision is reproducible from the input set at that commit.
G2
QA outputs are persisted
M1 grade flags, M2 K-value warnings, and M4 volume reports go to ./out/qa/ on every run for review.
G3
Idempotent re-runs
Every module deletes its previous output (via name or XData) before regenerating. No silent duplication, no orphan geometry.
G4
Human gates stay in place
Designer sign-off on QA report, checker review of BOQ, and PM approval of the published sheet set. Automation accelerates — it doesn't replace approval.
Failure mode to expect: CSV typos. Add a column validator at the top of each module (already done in M1/M2) and refuse to run if required columns are missing or non-numeric. Loud failure beats silent wrong geometry.
